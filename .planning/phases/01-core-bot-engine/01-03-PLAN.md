"""Simulate RPC gate — no trade proceeds without tesSUCCESS."""

import logging
from dataclasses import dataclass
from typing import Optional

from xrpl.clients import JsonRpcClient
from xrpl.models import Payment, Request

from src.config import XRPL_RPC_URL

logger = logging.getLogger(__name__)


@dataclass
class SimResult:
    """Result of a simulate RPC call."""
    success: bool
    result_code: str
    raw: Optional[dict] = None
    error: Optional[str] = None


def _build_rpc_client() -> JsonRpcClient:
    """Create a JSON-RPC client for simulate calls."""
    return JsonRpcClient(XRPL_RPC_URL)


async def simulate_transaction(tx_dict: dict, rpc_client: Optional[JsonRpcClient] = None) -> SimResult:
    """Run simulate RPC against live ledger. Returns SimResult.
    
    Only returns success=True if TransactionResult == "tesSUCCESS".
    Any other result or exception returns success=False.
    """
    import asyncio
    
    if rpc_client is None:
        rpc_client = _build_rpc_client()

    try:
        req = Request(command="simulate", tx_json=tx_dict, binary=False)
        response = await asyncio.to_thread(rpc_client.request, req)

        if not response.is_successful():
            return SimResult(
                success=False,
                result_code="rpc_error",
                error=str(response.result),
            )

        result = response.result
        tx_result = result.get("meta", {}).get("TransactionResult", "unknown")

        if tx_result == "tesSUCCESS":
            logger.info(f"Simulation passed: tesSUCCESS")
            return SimResult(success=True, result_code=tx_result, raw=result)
        else:
            logger.warning(f"Simulation failed: {tx_result}")
            return SimResult(success=False, result_code=tx_result, raw=result)

    except Exception as e:
        logger.error(f"Simulate RPC error: {e}")
        return SimResult(success=False, result_code="exception", error=str(e))
```

"""Trade executor — DRY_RUN branching, live execution, post-trade validation."""

import asyncio
import logging
from decimal import Decimal
from typing import Optional

from xrpl.clients import JsonRpcClient
from xrpl.models import Payment
from xrpl.transaction import autofill, sign
from xrpl.wallet import Wallet

from src.config import XRPL_SECRET, XRPL_RPC_URL, DRY_RUN
from src.pathfinder import Opportunity
from src.simulator import simulate_transaction, SimResult
from src.safety import CircuitBreaker, Blacklist
from src.trade_logger import log_trade
from src.telegram_alerts import send_alert

logger = logging.getLogger(__name__)

DROPS_PER_XRP = Decimal("1000000")


class TradeExecutor:
    """Executes or paper-trades opportunities after simulation gate."""

    def __init__(
        self,
        wallet: Wallet,
        circuit_breaker: CircuitBreaker,
        blacklist: Blacklist,
        rpc_client: Optional[JsonRpcClient] = None,
        dry_run: bool = DRY_RUN,
    ):
        self.wallet = wallet
        self.circuit_breaker = circuit_breaker
        self.blacklist = blacklist
        self.rpc_client = rpc_client or JsonRpcClient(XRPL_RPC_URL)
        self.dry_run = dry_run

    async def execute(self, opportunity: Opportunity) -> bool:
        """Process an opportunity through simulation gate and execute/log.
        
        Returns True if trade was executed (paper or live), False if skipped.
        """
        # Safety checks first
        if self.circuit_breaker.is_halted():
            logger.warning("Circuit breaker HALTED — skipping trade")
            return False

        if self.blacklist.is_blacklisted(opportunity.paths):
            logger.warning("Path is blacklisted — skipping trade")
            return False

        # Build Payment transaction
        input_drops = str(int(opportunity.input_xrp * DROPS_PER_XRP))
        output_drops = str(int(opportunity.output_xrp * DROPS_PER_XRP))
        send_max_drops = str(int(opportunity.input_xrp * DROPS_PER_XRP * Decimal("1.01")))

        tx = Payment(
            account=self.wallet.address,
            amount=output_drops,
            destination=self.wallet.address,
            paths=opportunity.paths,
            send_max=send_max_drops,
        )
        tx_dict = tx.to_dict()

        # SIMULATION GATE — must pass before any execution
        sim_result = await simulate_transaction(tx_dict, self.rpc_client)
        if not sim_result.success:
            logger.warning(
                f"Simulation FAILED ({sim_result.result_code}) — trade rejected"
            )
            return False

        # Build trade data for logging
        trade_data = {
            "profit_pct": str(opportunity.profit_pct),
            "profit_ratio": str(opportunity.profit_ratio),
            "input_xrp": str(opportunity.input_xrp),
            "output_xrp": str(opportunity.output_xrp),
            "simulated_output": str(opportunity.output_xrp),
            "dry_run": self.dry_run,
            "simulation_result": sim_result.result_code,
        }

        if self.dry_run:
            # DRY_RUN: log and alert, no submission
            msg = (
                f"DRY-RUN: {opportunity.profit_pct:.4f}% profit opportunity | "
                f"In: {opportunity.input_xrp} XRP -> Out: {opportunity.output_xrp} XRP"
            )
            logger.info(msg)
            await send_alert(msg)
            await log_trade(trade_data)
            return True

        # LIVE EXECUTION
        try:
            # autofill adds Sequence, Fee, LastLedgerSequence
            filled_tx = await asyncio.to_thread(autofill, tx, self.rpc_client)
            signed_tx = sign(filled_tx, self.wallet)

            # Submit signed transaction
            from xrpl.transaction import submit
            submit_response = await asyncio.to_thread(
                submit, signed_tx, self.rpc_client
            )
            submit_result = submit_response.result

            tx_hash = submit_result.get("tx_json", {}).get("hash", "unknown")
            engine_result = submit_result.get("engine_result", "unknown")

            trade_data["hash"] = tx_hash
            trade_data["engine_result"] = engine_result

            if engine_result == "tesSUCCESS":
                # Post-trade validation: check result matches simulation
                logger.info(
                    f"LIVE EXECUTED: {opportunity.profit_pct:.4f}% profit | Hash: {tx_hash}"
                )
                await send_alert(
                    f"LIVE TRADE: {opportunity.profit_pct:.4f}% profit | "
                    f"In: {opportunity.input_xrp} XRP -> Out: {opportunity.output_xrp} XRP | "
                    f"Hash: {tx_hash}"
                )
                # Record for circuit breaker
                profit_xrp = opportunity.output_xrp - opportunity.input_xrp
                self.circuit_breaker.record_trade(profit_xrp)
            else:
                # Failed submission — log full error details (LIVE-03)
                logger.error(
                    f"Live submission FAILED: {engine_result} | "
                    f"Full result: {submit_result}"
                )
                trade_data["error"] = str(submit_result)
                await send_alert(
                    f"TRADE FAILED: {engine_result} | "
                    f"Profit was {opportunity.profit_pct:.4f}%"
                )

            await log_trade(trade_data)
            return engine_result == "tesSUCCESS"

        except Exception as e:
            logger.error(f"Live execution error: {e}")
            trade_data["error"] = str(e)
            await log_trade(trade_data)
            await send_alert(f"EXECUTION ERROR: {e}")
            return False
```

"""XRPL 99%+ Arbitrage Bot — main entry point.

Connects to XRPL mainnet, scans for arbitrage opportunities each ledger close,
validates via simulate RPC, and either paper-trades or live-executes.
"""

import asyncio
import logging
from decimal import Decimal

from xrpl.wallet import Wallet

from src.config import (
    XRPL_SECRET,
    DRY_RUN,
    LOG_LEVEL,
    PROFIT_THRESHOLD,
    MAX_POSITION_PCT,
)
from src.connection import XRPLConnection
from src.pathfinder import PathFinder
from src.executor import TradeExecutor
from src.safety import CircuitBreaker, Blacklist
from src.trade_logger import setup_logging
from src.telegram_alerts import send_alert

logger = logging.getLogger(__name__)


async def main():
    """Main bot loop."""
    # Setup logging
    setup_logging()

    # Validate wallet secret
    if not XRPL_SECRET:
        logger.error("XRPL_SECRET not set in .env — cannot start bot")
        return

    wallet = Wallet.from_seed(XRPL_SECRET)
    logger.info(f"Wallet address: {wallet.address}")
    logger.info(f"Mode: {'DRY RUN (paper trading)' if DRY_RUN else 'LIVE TRADING'}")
    logger.info(f"Profit threshold: {PROFIT_THRESHOLD * 100}%")
    logger.info(f"Max position: {MAX_POSITION_PCT * 100}% of balance")

    # Initialize modules
    connection = XRPLConnection()
    pathfinder = PathFinder(connection, wallet.address)
    circuit_breaker = CircuitBreaker(account_address=wallet.address, connection=connection)
    blacklist = Blacklist()
    executor = TradeExecutor(
        wallet=wallet,
        circuit_breaker=circuit_breaker,
        blacklist=blacklist,
        dry_run=DRY_RUN,
    )

    # Track scan state
    scan_count = 0

    async def on_ledger_close(ledger_index: int):
        """Called every ~3-5 seconds when a new ledger closes."""
        nonlocal scan_count
        scan_count += 1

        if circuit_breaker.is_halted():
            if scan_count % 100 == 0:  # Log every ~5 minutes
                logger.info("Circuit breaker active — skipping scans")
            return

        try:
            # Get current balance for position sizing
            balance = await connection.get_account_balance(wallet.address)
            if balance <= Decimal("0"):
                logger.warning("Zero balance — skipping scan")
                return

            # Scan for opportunities
            opportunities = await pathfinder.scan(balance)

            for opp in opportunities:
                logger.info(
                    f"Opportunity #{scan_count}: {opp.profit_pct:.4f}% profit | "
                    f"In: {opp.input_xrp} XRP -> Out: {opp.output_xrp} XRP"
                )
                await executor.execute(opp)

            # Heartbeat logging every ~50 ledgers (~3 minutes)
            if scan_count % 50 == 0:
                logger.info(
                    f"Heartbeat: ledger={ledger_index}, scans={scan_count}, "
                    f"balance={balance} XRP, halted={circuit_breaker.is_halted()}"
                )

        except Exception as e:
            logger.error(f"Scan error at ledger {ledger_index}: {e}")

    # Register callback and connect
    connection.on_ledger_close(on_ledger_close)

    await send_alert(
        f"Bot started | Mode: {'DRY RUN' if DRY_RUN else 'LIVE'} | "
        f"Address: {wallet.address}"
    )

    logger.info("Starting XRPL connection...")
    await connection.connect()  # This runs forever with auto-reconnect


if __name__ == "__main__":
    asyncio.run(main())

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Bot -> XRPL RPC (simulate) | simulate RPC response determines if real money moves |
| Bot -> XRPL RPC (submit) | Signed transaction leaves the bot and enters the network |
| Wallet seed -> signed tx | Private key used to sign transactions |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
